02. Реализация на Python

Использование объектно-ориентированного подхода для описания финансовых функций.

class FinancialModel:
    def __init__(self, principal, rate, time):
        self.p = principal
        self.r = rate / 100
        self.t = time

    def get_simple(self):
        return self.p * (1 + self.r * self.t)

    def get_compound(self, m=12):
        return self.p * (1 + self.r/m)**(m * self.t)

# Расчет для суммы 100,000 под 15% на 5 лет
model = FinancialModel(100000, 15, 5)
print(model.get_compound())
